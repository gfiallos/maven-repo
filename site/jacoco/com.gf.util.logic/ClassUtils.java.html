package com.gf.util.logic;

import com.gf.util.logic.exception.GenericException;
import com.gf.util.model.dto.crud.FieldData;
import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import java.util.Arrays;
import java.util.List;
import java.util.Objects;
import java.util.stream.Collectors;

/**
 * The type Class utils.
 */
public final class ClassUtils {
  private static final Logger LOG = LogManager.getLogger(ClassUtils.class);

  /**
   * Instantiates a new Class utils.
   */
  private ClassUtils() {
  }

  /**
   * List fields list.
   *
   * @param pClass the p class
   * @return the list
   */
  public static List<FieldData> listFields(Class<?> pClass) {
    return Arrays.stream(pClass.getDeclaredFields())
                 .filter(field -> !Objects.equals(field.getName(), "serialVersionUID"))
                 .filter(field -> !field.getName()
                                        .contains("$"))
                 .filter(field -> !field.getType()
                                        .getName()
                                        .contains("List"))
                 .map(field -> new FieldData().setName(field.getName())
                                              .setType(field.getType()
                                                            .getSimpleName()))
                 .collect(Collectors.toList());
  }

  /**
   * Create instance z.
   *
   * @param <Z>    the type parameter
   * @param pClass the p class
   * @return the z
   */
  public static <Z> Z createInstance(Class<Z> pClass) {
    try {
      return pClass.getConstructor()
                   .newInstance();
    } catch (Exception e) {
      LOG.warn(e, e);
      throw new GenericException(e.getMessage(), e);
    }
  }
}
