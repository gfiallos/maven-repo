package com.gf.util.logic;

import java.util.Optional;
import java.util.function.UnaryOperator;

/**
 * The type String util.
 * <p>
 * Provides some useful methods for String
 */
public final class StringUtil {
  /**
   * Instantiates a new String util.
   */
  private StringUtil() {
  }

  /**
   * Capitalize string.
   *
   * @param pData the p data
   * @return the string
   */
  public static String capitalize(String pData) {
    return manage(pData, StringUtil::manageCapitalize);
  }

  /**
   * Manage string.
   *
   * @param pData   the p data
   * @param pMapper the p mapper
   * @return the string
   */
  private static String manage(String pData, UnaryOperator<String> pMapper) {
    return Optional.ofNullable(pData)
                   .map(pMapper)
                   .orElse(pData);
  }

  /**
   * Manage capitalize string.
   *
   * @param pData the p data
   * @return the string
   */
  private static String manageCapitalize(String pData) {
    var c = pData.toCharArray();
    c[0] = Character.toUpperCase(c[0]);
    return String.valueOf(c);
  }

  /**
   * Decapitalize string.
   *
   * @param pData the p data
   * @return the string
   */
  public static String decapitalize(String pData) {
    return manage(pData, StringUtil::manageDecapitalize);
  }

  /**
   * Manage decapitalize string.
   *
   * @param pData the p data
   * @return the string
   */
  private static String manageDecapitalize(String pData) {
    var c = pData.toCharArray();
    c[0] = Character.toLowerCase(c[0]);
    return String.valueOf(c);
  }

  /**
   * Split camel case string.
   *
   * @param pData the p data
   * @return the string
   */
  public static String splitCamelCase(String pData) {
    return pData.replaceAll(
            String.format("%s|%s|%s",
                    "(?<=[A-Z])(?=[A-Z][a-z])",
                    "(?<=[^A-Z])(?=[A-Z])",
                    "(?<=[A-Za-z])(?=[^A-Za-z])"
            ),
            " "
    );
  }
}
