package com.gf.util.logic;

import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import java.security.NoSuchAlgorithmException;
import java.security.SecureRandom;
import java.util.Optional;
import java.util.Random;

/**
 * The type Random util.
 */
public class RandomUtil {
  /**
   * The constant LOG.
   */
  private static final Logger LOG = LogManager.getLogger(RandomUtil.class);
  /**
   * The Rand.
   */
  private Random rand;

  /**
   * Instantiates a new Random util.
   */
  public RandomUtil() {
    try {
      this.rand = SecureRandom.getInstanceStrong();
    } catch (NoSuchAlgorithmException pE) {
      LOG.warn("Could not get a secure random number generator", pE);
      this.rand = null;
    }
  }

  /**
   * Generate random from min and max params.
   *
   * @param pMin the p min
   * @param pMax the p max
   * @return the int
   */
  public int generateRandomInt(int pMin, int pMax) {
    return Optional.ofNullable(this.rand)
                   .map(r -> r.nextInt(pMax - pMin + 1) + pMin)
                   .orElse(0);

  }
}
