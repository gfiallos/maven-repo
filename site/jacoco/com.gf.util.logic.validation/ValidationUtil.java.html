package com.gf.util.logic.validation;


import javax.validation.Validation;
import java.util.List;
import java.util.stream.Collectors;

/**
 * The type Validation util.
 */
public final class ValidationUtil {
  /**
   * Instantiates a new Validation util.
   */
  private ValidationUtil() {
  }

  /**
   * Validate the object based in javax.validation Framework.
   *
   * @param pObject the p object
   * @return the list
   */
  public static List<String> validate(Object pObject) {
    return Validation.buildDefaultValidatorFactory()
                     .getValidator()
                     .validate(pObject)
                     .stream()
                     .map(v -> v.getPropertyPath()
                                .toString() + " " + v.getMessage())
                     .collect(Collectors.toList());
  }
}
