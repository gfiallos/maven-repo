package com.gf.logic.cache;

import org.ehcache.Cache;
import org.ehcache.CacheManager;
import org.ehcache.config.builders.CacheConfigurationBuilder;
import org.ehcache.config.builders.CacheManagerBuilder;
import org.ehcache.config.builders.ExpiryPolicyBuilder;
import org.ehcache.config.builders.ResourcePoolsBuilder;

import java.time.Duration;
import java.util.Optional;

/**
 * The type Cache manager util.
 */
public final class CacheManagerUtil {
  /**
   * The constant CACHE_ALIAS.
   */
  public static final String CACHE_ALIAS = "cacheAlias";
  /**
   * The constant CACHE_REGION.
   */
  public static final String CACHE_REGION = "cacheRegion";
  /**
   * The constant DEFAULT_DURATION.
   */
  private static final int DEFAULT_DURATION = 120;
  /**
   * The constant ENTRIES_HEAP_SIZE.
   */
  private static final int ENTRIES_HEAP_SIZE = 100;
  /**
   * The constant cacheManager.
   */
  private static CacheManager cacheManager;

  /**
   * Instantiates a new Cache manager util.
   */
  private CacheManagerUtil() {
  }

  /**
   * Build cache cache.
   *
   * @param pTime the p time
   * @return the cache
   */
  private static Cache<String, Object> buildCache(Integer pTime) {
    return CacheManagerUtil.cacheManager
            .createCache(CacheManagerUtil.CACHE_REGION, CacheConfigurationBuilder
                    .newCacheConfigurationBuilder(String.class, Object.class, ResourcePoolsBuilder.heap(ENTRIES_HEAP_SIZE))
                    .withExpiry(ExpiryPolicyBuilder.timeToLiveExpiration(Duration.ofMinutes(pTime))));
  }

  /**
   * Clean cache.
   */
  public static void cleanCache() {
    synchronized (CacheManagerUtil.class) {
      Optional.ofNullable(CacheManagerUtil.cacheManager)
              .ifPresent(r -> r.removeCache(CacheManagerUtil.CACHE_REGION));
    }
  }

  /**
   * Creates if it is needed and Gets the unique instance of this class.
   *
   * @param pTime the p time
   * @return the instance
   */
  public static CacheHelper createInstance(Integer pTime) {
    CacheManagerUtil.prepareManager();
    return new CacheHelper(CacheManagerUtil.buildCache(pTime));
  }

  /**
   * Prepare manager.
   */
  private static void prepareManager() {
    if (CacheManagerUtil.cacheManager != null) {
      return;
    }
    var cacheConfiguration = CacheConfigurationBuilder.newCacheConfigurationBuilder(String.class,
                                                              Object.class,
                                                              ResourcePoolsBuilder.heap(ENTRIES_HEAP_SIZE))
                                                      .withExpiry(ExpiryPolicyBuilder.timeToLiveExpiration(Duration.ofSeconds(DEFAULT_DURATION)))
                                                      .build();
    CacheManagerUtil.cacheManager = CacheManagerBuilder.newCacheManagerBuilder()
                                                       .withCache(CacheManagerUtil.CACHE_ALIAS, cacheConfiguration)
                                                       .build();
    CacheManagerUtil.cacheManager.init();
  }
}
