package com.gf.logic.cache;


import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;
import org.ehcache.Cache;
import org.springframework.data.util.CastUtils;

import java.util.List;
import java.util.Optional;
import java.util.function.Supplier;
import java.util.stream.Collectors;
import java.util.stream.StreamSupport;


/**
 * Utility Class Cache helper.
 */
public class CacheHelper {


  /**
   * The constant LOG.
   */
  private static final Logger LOG = LogManager.getLogger(CacheHelper.class);

  /**
   * The Cache.
   */
  private final Cache<String, Object> cache;

  /**
   * Instantiates a new Cache helper.
   *
   * @param pCache the p cache
   */
  public CacheHelper(Cache<String, Object> pCache) {
    CacheHelper.LOG.info("Initializing Cache");
    this.cache = pCache;
  }

  /**
   * Instantiates a new Cache helper.
   */
  public CacheHelper() {
    this.cache = null;
  }

  /**
   * Clean cache.
   */
  public void cleanCache() {
    this.cache.clear();
  }

  /**
   * Contains key boolean.
   *
   * @param pKey the p key
   * @return the boolean
   */
  public boolean containsKey(String pKey) {
    return this.cache.containsKey(pKey);
  }

  /**
   * Gets value.
   *
   * @param <T>   the type parameter
   * @param pKey  The key value
   * @param pType The type value
   * @return the value
   */
  public <T> Optional<T> getValue(String pKey, Class<T> pType) {
    CacheHelper.LOG.debug("Optional {}", pType);
    return CastUtils.cast(this.prepareCache()
                              .map(c -> (c.containsKey(pKey)) ? c.get(pKey) : null));

  }

  /**
   * Gets value.
   *
   * @param <T>       the type parameter
   * @param pKey      the p key
   * @param pType     the p type
   * @param pSupplier the p supplier
   * @return the value
   */
  public <T> T getValue(String pKey, Class<T> pType, Supplier<T> pSupplier) {
    var val = this.getValue(pKey, pType)
                  .orElseGet(pSupplier);
    this.putValue(pKey, val);
    return val;
  }

  /**
   * Gets value.
   *
   * @param pKey the key value
   * @return the value
   */
  public Optional<Object> getValue(String pKey) {
    return Optional.ofNullable(this.cache.get(pKey));
  }

  /**
   * List keys list.
   *
   * @return the list
   */
  public List<String> listKeys() {
    return StreamSupport.stream(this.cache.spliterator(), false)
                        .map(Cache.Entry::getKey)
                        .collect(Collectors.toList());
  }

  /**
   * Prepare cache optional.
   *
   * @return the optional
   */
  private Optional<Cache<String, Object>> prepareCache() {
    return Optional.ofNullable(this.cache);
  }

  /**
   * Put value t.
   *
   * @param <T>    the type parameter
   * @param pKey   The key value
   * @param pValue The value value
   * @return the t
   */
  public <T> T putValue(String pKey, T pValue) {
    this.prepareCache()
        .ifPresent(c -> c.put(pKey, pValue));

    return pValue;
  }

  /**
   * Remove value.
   *
   * @param pKey the key value
   */
  public void removeValue(String pKey) {
    this.cache.remove(pKey);
  }
}
