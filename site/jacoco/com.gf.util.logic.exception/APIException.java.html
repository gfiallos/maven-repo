package com.gf.util.logic.exception;

import java.io.Serializable;
import java.util.Map;

/**
 * Exception Api exception used to Manage API Errors.
 */
public class APIException extends GenericException {
  /**
   * The constant serialVersionUID.
   */
  private static final long serialVersionUID = 5287580138697720687L;
  /**
   * The Http status code.
   */
  private final int httpStatusCode;
  /**
   * The Exception data.
   */
  private final Map<String, Serializable> exceptionData;

  /**
   * Instantiates a new Api exception.
   *
   * @param pExceptionData the p exception data
   */
  public APIException(Map<String, Serializable> pExceptionData) {
    super("" + pExceptionData.get("userMessage"));
    var data = pExceptionData;
    if (pExceptionData.containsKey("data")) {
      try {
        data = (Map<String, Serializable>) pExceptionData.get("data");
      } catch (Exception e) {
        //ignored
      }
    }
    this.exceptionData = data;
    this.httpStatusCode = Integer.parseInt("" + pExceptionData.computeIfAbsent("httpStatus", v -> "400"));
  }

  /**
   * Gets exception data.
   *
   * @return the exception data
   */
  public Map<String, Serializable> getExceptionData() {
    return this.exceptionData;
  }

  /**
   * Gets http status code.
   *
   * @return the http status code
   */
  public int getHttpStatusCode() {
    return this.httpStatusCode;
  }
}
