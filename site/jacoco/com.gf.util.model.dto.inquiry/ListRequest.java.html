package com.gf.util.model.dto.inquiry;

import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
import com.gf.util.model.domain.Operator;
import lombok.Data;

import javax.validation.Valid;
import javax.validation.constraints.NotNull;
import java.util.ArrayList;
import java.util.List;
import java.util.Optional;

/**
 * The type List request.
 * <p>
 * Used to request a list of entities, considering the filters and the pagination.
 */
@Data
@JsonIgnoreProperties(value = {"parameters"})
public class ListRequest {
  /**
   * The Criteria.
   */
  @Valid
  private List<Criteria> criteria;
  /**
   * The Order.
   */
  @Valid
  private List<OrderRequest> order;
  /**
   * The Pagination.
   */
  @NotNull
  @Valid
  private Pagination pagination;
  /**
   * The Parameters.
   */
  @Valid
  private List<Parameter> parameters;

  /**
   * Add criteria list request.
   *
   * @param pName  the p name
   * @param pValue the p value
   * @return the list request
   */
  public ListRequest addCriteria(String pName, Object pValue) {
    return this.addCriteria(pName, pValue, Operator.equals);
  }

  /**
   * Add criteria list request.
   *
   * @param pName     the p name
   * @param pValue    the p value
   * @param pOperator the p operator
   * @return the list request
   */
  public ListRequest addCriteria(String pName, Object pValue, Operator pOperator) {
    Optional.ofNullable(this.criteria)
            .orElseGet(() -> this.criteria = new ArrayList<>())
            .add(new Criteria().setName(pName)
                               .setOperator(pOperator)
                               .setValue(pValue));
    return this;
  }

  /**
   * Add parameter list request.
   *
   * @param pName  the p name
   * @param pValue the p value
   * @return the list request
   */
  public ListRequest addParameter(String pName, Object pValue) {
    Optional.ofNullable(this.parameters)
            .orElseGet(() -> this.parameters = new ArrayList<>())
            .add(new Parameter().setName(pName)
                                .setValue(pValue));
    return this;
  }

  /**
   * Find criteria string.
   *
   * @param pName the p name
   * @return the string
   */
  public Object findCriteria(String pName) {
    return Optional.ofNullable(this.getCriteria())
                   .orElseGet(ArrayList::new)
                   .stream()
                   .filter(c1 -> c1.getName()
                                   .equals(pName))
                   .findFirst()
                   .map(Criteria::getValue)
                   .orElse(null);
  }

  /**
   * Gets parameter.
   *
   * @param <T>      the type parameter
   * @param pName    the p name
   * @param pDefault the p default
   * @return the parameter
   */
  public <T> T getParameter(String pName, T pDefault) {
    return (T) Optional.ofNullable(this.parameters)
                       .map(ps -> ps.stream()
                                    .filter(f -> f.getName()
                                                  .equals(pName))
                                    .findFirst()
                                    .map(Parameter::getValue)
                                    .orElse(pDefault))
                       .orElse(pDefault);
  }

  /**
   * Has criteria boolean.
   *
   * @param pName the p name
   * @return the boolean
   */
  public boolean hasCriteria(String pName) {
    return Optional.ofNullable(this.getCriteria())
                   .map(c -> c.stream()
                              .anyMatch(c1 -> c1.getName()
                                                .equals(pName)))
                   .orElse(false);
  }
}
